/**
* 功能：仅判断当前是否是微信
* @param {Any} - 并不需要任何参数
* @returns {Boolean} - 如果是微信浏览器，会返回true反则会返回false
*/

function is_wx(){
    const ua = window.navigator.userAgent.toLowerCase();
    if(ua.match(/MicroMessenger/i) === 'micromessenger'){
        return true
    }
    return false
}
export default is_wx;
