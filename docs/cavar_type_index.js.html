/**
* 功能：判断数据类型功能函数
* @param {Any} currTask - 判断类型的格式
* @returns {String} - 返回传入的参数类型，以小写字符串的形式返回
*/
function cavar_type(currTask) {
    return Object.prototype.toString
      .call(currTask)
      .slice(8, -1)
      .toLocaleLowerCase();
  }
export default cavar_type
