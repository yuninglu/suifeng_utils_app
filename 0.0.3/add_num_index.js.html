import cavar_type from '../cavar_type'
/**
* 功能： 两个数相加
* @param {number} augend - 被加数
* @param {number} addend - 加数
* @returns {number} - 两数相加的结果值
*/
function add_num(augend, addend) {
    if (cavar_type(augend) === "number" && cavar_type(addend) === "number") {
      if (!isNaN(augend) && !isNaN(addend)) {
        return augend + addend;
      } else {
        console.error(new Error("add_num: arguments is not NaN, you is sb"));
      }
    } else {
      console.error(new Error("add_num: arguments is not number, you is sb"));
    }
  }
  export default add_num;
